# --- DEPLOYMENT ---
# 1. Sync .env to VPS (Required for Clerk/Auth to work on production)
scp .env ct@66.179.241.25:~/realestateoportunity/

# 2. Build & Push images (from localhost for VPS platform)
# Load variables from .env for the build args (Mac/Linux compatible)
# Note: Ensure you are in the project root where .env exists
docker buildx build --platform linux/amd64 \
  --build-arg VITE_CLERK_PUBLISHABLE_KEY=$(grep VITE_CLERK_PUBLISHABLE_KEY .env | cut -d '=' -f2) \
  -t carlostrevisan/realestate-frontend:latest --push ./client --target production

docker buildx build --platform linux/amd64 -t carlostrevisan/realestate-backend:latest --push ./server --target production
docker buildx build --platform linux/amd64 -t carlostrevisan/realestate-data-worker:latest --push ./data-engine

# 3. Update VPS
git pull

# Option A: Standard Update (Keep existing data)
ssh ct@66.179.241.25 "cd ~/realestateoportunity && docker compose pull && docker compose up -d"
